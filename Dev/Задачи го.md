 ```go
		}

		}

			ch3 <- i
		}
		close(ch3)
	}()

		println(id)
	}

}

func merge(chans ...chan int) chan int {
	out := make(chan int)
	wg := sync.WaitGroup{}

	for _, ch := range chans {
		wg.Add(1)
		go func(ch chan int) {
			defer wg.Done()
			for c := range ch {
				out <- c
			}
		}(ch)
	}

	go func() {
		wg.Wait()
		close(out)
	}()

	return out
}
 ```

```go
func longestSubArray(nums []int) int {
	zeroes, res, start := 0, 0, 0

	for end, num := range nums {
		if num == 0 {
			zeroes++
		}
		for zeroes > 1 {
			if nums[start] == 0 {
				zeroes--
			}
			start++
		}
		res = max(res, end-start)
	}

	return res
}
```

```go
func isAnagram(s, t string) bool {
	lettersMap := make(map[rune]int)

	for _, letter := range s {
		lettersMap[letter]++
	}
	for _, letter := range t {
		lettersMap[letter]--
		if lettersMap[letter] < 0 {
			return false
		}
	}
	for _, count := range lettersMap {
		if count != 0 {
			return false
		}
	}
	return true
```

```go
func mergeSort(a, b []int) []int {
	result := make([]int, 0, len(a)+len(b))

	i, j := 0, 0

	for i < len(a) || j < len(b) {
		if i == len(a) {
			result = append(result, b[j:]...)
			break
		}
		if j == len(b) {
			result = append(result, a[i:]...)
			break
		}

		if a[i] < b[j] {
			result = append(result, a[i])
			i++
		} else {
			result = append(result, b[j])
			j++
		}
	}
	return result
}
```

```go
func mergeUnSort(a, b []int) []int {
	result := make([]int, 0, len(a)+len(b))

	i, j := 0, 0

	if len(a) == 0 {
		return b
	}

	if len(b) == 0 {
		return a
	}

	for i < len(a) && j < len(b) {
		if a[i] < b[j] {
			result = append(result, a[i])
			i++
		} else if a[i] > b[j] {
			result = append(result, b[j])
			j++
		} else {
			if len(result) > 0 && result[len(result)-1] != a[i] {
				result = append(result, a[i])
			}
			i++
			j++
		}
	}

	if i == len(a) {
		for j < len(b) {
			if len(result) > 0 && result[len(result)-1] != b[j] {
				result = append(result, b[j])
			}
			j++
		}
	}
	if j == len(b) {
		for i < len(a) {
			if len(result) > 0 && result[len(result)-1] != a[i] {
				result = append(result, a[i])
			}
			i++
		}
	}

	if len(result) == 0 {
		if len(a) != 0 {
			result = append(result, a[0])
		} else {
			result = append(result, b[0])
		}
	}

	return result
}
```

# Структуры и Интерфейсы

Кастинг (приведение типов) структуры из одного интерфейса в другой осуществляется через промежуточное приведение к конкретному типу (структуре).
В языках с сильной типизацией (например, Go, C#), интерфейс — это лишь набор методов. Чтобы получить доступ к методам другого интерфейса, нужно сначала «достать» исходную структуру, а затем привести её к новому интерфейсу.

```go
type InterfaceA interface {
    MethodA()
}

type InterfaceB interface {
    MethodB()
}

type MyStruct struct{}

func (s MyStruct) MethodA() { fmt.Println("A") }
func (s MyStruct) MethodB() { fmt.Println("B") }

func main() {
    var ia InterfaceA = MyStruct{}

    // 1. Кастим InterfaceA -> MyStruct
    myConcreteStruct := ia.(MyStruct)

    // 2. Кастим MyStruct -> InterfaceB
    var ib InterfaceB = myConcreteStruct
    
    ib.MethodB()
}
```

Интерфейс представляет собой пару (тип, значение). Если у интерфейса задан тип, но значение равно `nil`, интерфейс **не** равен `nil`

Основные случаи, когда интерфейс равен nil:
- Нулевая инициализация: Когда переменная интерфейса объявлена, но ей не присвоено значение (например, var i interface{}), она по умолчанию равна nil.
- Явное присваивание: Когда интерфейсу явно присваивается nil (i = nil).
- Присваивание типизированного nil: Когда переменной интерфейса присваивается указатель, который равен nil (например, `var p *int = nil; i = p`), интерфейс станет nil. 

Когда интерфейс НЕ равен nil (ловушка):
- Если в интерфейс завернут типизированный nil (например, указатель на структуру `*MyStruct(nil)`), интерфейс хранит тип `*MyStruct` и значение nil. В этом случае проверка if i == nil вернет false. 

Интерфейс содержит тип, который реализует и таблицу методов и мы тут заполнили одно из полей
```go
type SomeStruct struct {
	Value int
}

func CheckForNil(i interface{}) {
	if i == nil {
		fmt.Println("it's is nil")
		return
	}
	fmt.Println("it's not nil")
}

func main() {
	var s *SomeStruct
	CheckForNil(s)
}
```

# Каналы

Запись данных в канал
- Открытый канал
	- Буферизированный: Запись проходит успешно, если в буфере есть место. Если буфер полон, горутина блокируется до тех пор, пока данные не будут прочитаны.
	- Небуферизированный: Запись блокируется до тех пор, пока другая горутина не начнет чтение из этого канала. 
- Закрытый канал
	- Запись в закрытый канал вызывает panic (аварийное завершение программы).
- Nil-канал
	- Запись в nil-канал (канал, который был объявлен, но не инициализирован через make) блокирует горутину навсегда.

```go
	ch := make(chan bool)
	go func() {
		time.Sleep(3 * time.Second) 
		ch <- false
	}()

	ticker := time.NewTicker(time.Second)

		case <-ticker.C:

			ch <- true // Зависли в ожидании читателя из канала
		case value := <-ch:
			fmt.Printf("Channel received %t\n", value)
			return
		}
	}
```

```go
func requestData(timeout time.Duration) string {
	c := make(chan string)

	go func() {
		c <- requestFromSlowService()
	}()
	select {
	case result := <-c:
		fmt.Printf("Request returned: %s", result)
		return result
	case <-time.After(timeout):
		fmt.Printf("Request timeout\n")
		return "empty val"
	}
}
```
